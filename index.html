# Simple EPS computation with buybacks effect
import math

net_income = 1_020_000_000   # USD
shares_start = 720_000_000
buyback_shares = 20_000_000
avg_shares = shares_start - buyback_shares/2

basic_eps = net_income / avg_shares
print(f"EPS: {basic_eps:.2f}")
